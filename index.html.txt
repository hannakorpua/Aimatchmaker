  const g=document.getElementById('contentGrid'); g.innerHTML='';
  pool.forEach(item=>{
    const d=document.createElement('div'); d.className='ccard';
    d.innerHTML=`
      ${item.hot?'<div class="tag hot">🔥 Trending</div>':''}
      ${item.trap?'<div class="tag trap">👀 Looks relevant?</div>':''}
      <div class="cicon">${item.icon}</div>
      <div class="ctitle">${item.title}</div>
      <div class="csub">${item.sub}</div>`;
    d.onclick=()=>{
      if(d.classList.contains('sel')){
        d.classList.remove('sel');
        picks=picks.filter(x=>x!==item.id);
      } else {
        if(picks.length>=3)return;
        d.classList.add('sel');
        picks.push(item.id);
      }
      updatePickStatus();
    };
    g.appendChild(d);
  });
}

function updatePickStatus(){
  const n=picks.length;
  [1,2,3].forEach(i=>document.getElementById('d'+i).classList.toggle('on',i<=n));
  document.getElementById('selTxt').textContent=
    n===0?'Pick at least 2':n===1?'Pick 1 more':n===2?'Good! Pick one more or continue':'3 selected — ready!';
  const b=document.getElementById('btnC');
  if(n>=2){b.className='btn btn-main';b.textContent='See what the AI picked →';}
  else{b.className='btn btn-disabled';b.textContent='Select '+(2-n)+' more card'+(2-n>1?'s':'');}
}

// ── S4: Reveal ────────────────────────────────────────────────────────────
function buildReveal(){
  aiRecs=AI_MAP[profile.id];
  aiRecs.forEach(id=>{G.trend[id]=(G.trend[id]||0)+1;});

  const matched=picks.filter(id=>aiRecs.includes(id)).length;
  const total=aiRecs.length;
  const pct=Math.round(matched/total*100);

  // Score banner
  const {col,emoji,title,msg}=pct===100
    ?{col:'\#32d74b',emoji:'🎯',title:'Perfect!',msg:`You got all ${total} of the AI's picks right`}
    :pct>=50
    ?{col:'\#ffd60a',emoji:'👍',title:'Not bad!',msg:`You matched ${matched} of ${total} AI picks`}
    :{col:'\#ff375f',emoji:'🤖',title:'Algorithm wins!',msg:`Only ${matched} of your guesses matched`};

  document.getElementById('scoreBanner').innerHTML=`
    <div class="score" style="background:${col}18;border:2px solid ${col}44;margin-bottom:16px">
      <div class="score-icon">${emoji}</div>
      <div class="score-txt"><h3 style="color:${col}">${title}</h3><p>${msg}</p></div>
    </div>`;

  // Cards: your picks first
  let html='<p style="font-size:.72rem;font-weight:700;letter-spacing:1px;text-transform:uppercase;color:var(--txt2);margin-bottom:8px">YOUR PICKS</p>';
  picks.forEach(id=>{
    const item=CONTENT.find(x=>x.id===id);
    const isMatch=aiRecs.includes(id);
    html+=`<div class="rev-card ${isMatch?'match':'miss'}">
      <div class="rev-icon">${item.icon}</div>
      <div class="rev-body">
        <div class="rev-title">${item.title}</div>
        <div class="rev-reason">${isMatch?REASONS[id]:'The algorithm skipped this one'}</div>
        <span class="badge ${isMatch?'g':'r'}">${isMatch?'✓ AI also picked this':'✗ Not in the AI\'s feed'}</span>
      </div></div>`;
  });

  // AI picks not in user's picks
  const missed=aiRecs.filter(id=>!picks.includes(id));
  if(missed.length){
    html+='<p style="font-size:.72rem;font-weight:700;letter-spacing:1px;text-transform:uppercase;color:var(--txt2);margin:14px 0 8px">AI ALSO PICKED</p>';
    missed.forEach(id=>{
      const item=CONTENT.find(x=>x.id===id);
      html+=`<div class="rev-card ai-only">
        <div class="rev-icon">${item.icon}</div>
        <div class="rev-body">
          <div class="rev-title">${item.title}</div>
          <div class="rev-reason">${REASONS[id]}</div>
          <span class="badge y">✨ Algorithm pick you missed</span>
        </div></div>`;
    });
  }
  document.getElementById('revCards').innerHTML=html;

  // Trending
  const sorted=Object.entries(G.trend).sort((a,b)=>b[1]-a[1]).slice(0,4);
  const tmax=sorted[0]?sorted[0][1]:1;
  let th=`<div class="trend-box"><div class="trend-head">🔥 Most picked today by all visitors</div>`;
  sorted.forEach(([id,cnt])=>{
    const item=CONTENT.find(x=>x.id===id)||{icon:'❓',title:id};
    th+=`<div class="trow"><span class="tlbl">${item.icon} ${item.title}</span><div class="tbar"><div class="tfill" style="width:${Math.round(cnt/tmax*100)}%"></div></div><span class="tcnt">${cnt}</span></div>`;
  });
  th+='</div>';
  document.getElementById('trendWrap').innerHTML=th;
}

  document.getElementById('btnYes').disabled = true;

}
